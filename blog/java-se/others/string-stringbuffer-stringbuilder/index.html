String StringBuffer StringBuilder

2018-07-18

дёүиҖ…зҡ„иҜҙжҳҺ

зұ»жҳҜеҗҰеҸҜеҸҳжҳҜеҗҰзәҝзЁӢе®үе…Ёж•ҲзҺҮ
StringеҜ№иұЎдёҚеҸҜеҸҳзәҝзЁӢе®үе…Ё3
StringBuilderеҜ№иұЎеҸҜеҸҳйқһзәҝзЁӢе®үе…Ё1
StringBufferеҜ№иұЎеҸҜеҸҳзәҝзЁӢе®үе…Ё2

StringBuffer е’Ң StringBuilder зҡ„е…¬е…ұзҲ¶зұ» AbstractStringBuilder
StringBuffer зҡ„еҗҢжӯҘжҳҜйҖҡиҝҮеңЁж–№жі•дёҠйқўж·»еҠ  synchronized жқҘе®һзҺ°гҖӮ

String

StringеҜ№иұЎзҡ„д»»дҪ•дҝ®ж”№йғҪдёҚдјҡеҪұе“ҚеҺҹеҜ№иұЎпјҢд»»дҪ•зӣёе…ізҡ„ж“ҚдҪңйғҪдјҡз”ҹжҲҗж–°зҡ„еҜ№иұЎгҖӮ дҫӢеӯҗпјҡ

public String substring(int beginIndex) {
        if (beginIndex < 0) {
            throw new StringIndexOutOfBoundsException(beginIndex);
        }
        int subLen = value.length - beginIndex;
        if (subLen < 0) {
            throw new StringIndexOutOfBoundsException(subLen);
        }
        return (beginIndex == 0) ? this : new String(value, beginIndex, subLen);
    }

String зҡ„ substring ж–№жі•пјҢиҝ”еӣһзҡ„ж–°зҡ„еҜ№иұЎжҳҜйҖҡиҝҮnewдәҶдёҖдёӢж–°зҡ„Stringзҡ„еҜ№иұЎиҝ”еӣһгҖӮеҰӮжһңејҖе§Ӣзҡ„index=0иҝ”еӣһзҡ„жҳҜеҪ“еүҚзҡ„еҜ№иұЎ

дҪҝз”Ёзҡ„еҺҹеҲҷ

  1. еҹәжң¬еҺҹеҲҷпјҡеҰӮжһңиҰҒж“ҚдҪңе°‘йҮҸзҡ„ж•°жҚ®пјҢз”ЁString пјӣеҚ•зәҝзЁӢж“ҚдҪңеӨ§йҮҸж•°жҚ®пјҢз”Ё StringBuilder пјӣеӨҡзәҝзЁӢж“ҚдҪңеӨ§йҮҸж•°жҚ®пјҢз”Ё StringBufferгҖӮ
  2. дёҚиҰҒдҪҝз”ЁStringзұ»зҡ„вҖқ+вҖқжқҘиҝӣиЎҢйў‘з№Ғзҡ„жӢјжҺҘпјҢеӣ дёәйӮЈж ·зҡ„жҖ§иғҪжһҒе·®зҡ„(жҜҸж¬ЎйғҪиҰҒз”ҹжҲҗж–°зҡ„зұ»)дҫӢеӯҗеҰӮдёӢпјҡ
String result = "";
for (String s : hugeArray) {
result = result + s;
}
// дҪҝз”ЁStringBuilder
StringBuilder sb = new StringBuilder();
for (String s : hugeArray) {
sb.append(s);
}
String result = sb.toString();

иҝҷж ·зҡ„жғ…еҶөжҳҫз„¶з”Ё StringBulider жҜ”иҫғеҘҪ 3. StringBuilder дёҖиҲ¬дҪҝз”ЁеңЁж–№жі•еҶ…йғЁжқҘе®ҢжҲҗзұ»дјјвҖқ+вҖқеҠҹиғҪпјҢеӣ дёәжҳҜзәҝзЁӢдёҚе®үе…Ёзҡ„пјҢжүҖд»Ҙз”Ёе®Ңд»ҘеҗҺеҸҜд»ҘдёўејғгҖӮ StringBuffer дё»иҰҒз”ЁеңЁе…ЁеұҖеҸҳйҮҸдёӯ 4. зӣёеҗҢжғ…еҶөдёӢдҪҝз”Ё StirngBuilder зӣёжҜ”дҪҝз”Ё StringBuffer д»…иғҪиҺ·еҫ— 10%~15% е·ҰеҸізҡ„жҖ§иғҪжҸҗеҚҮпјҢдҪҶеҚҙиҰҒеҶ’еӨҡзәҝзЁӢдёҚе®үе…Ёзҡ„йЈҺйҷ©гҖӮиҖҢеңЁзҺ°е®һзҡ„жЁЎеқ—еҢ–зј–зЁӢдёӯпјҢиҙҹиҙЈжҹҗдёҖжЁЎеқ—зҡ„зЁӢеәҸе‘ҳдёҚдёҖе®ҡиғҪжё…жҷ°ең°еҲӨж–ӯиҜҘжЁЎеқ—жҳҜеҗҰдјҡж”ҫе…ҘеӨҡзәҝзЁӢзҡ„зҺҜеўғдёӯиҝҗиЎҢпјҢеӣ жӯӨпјҡйҷӨйқһзЎ®е®ҡзі»з»ҹзҡ„з“¶йўҲжҳҜеңЁ StringBuffer дёҠпјҢе№¶дё”зЎ®е®ҡдҪ зҡ„жЁЎеқ—дёҚдјҡиҝҗиЎҢеңЁеӨҡзәҝзЁӢжЁЎејҸдёӢпјҢжүҚеҸҜд»ҘйҮҮз”Ё StringBuilder пјӣеҗҰеҲҷиҝҳжҳҜз”Ё StringBuffer
